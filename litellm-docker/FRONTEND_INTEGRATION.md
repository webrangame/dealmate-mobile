## 🎯 Overview

LiteLLM acts as a proxy that:
- Manages multiple LLM providers (OpenAI, Anthropic, Google Gemini, etc.)
- Handles user authentication and API keys
- Tracks usage, spending, and rate limits
- Provides a unified API interface

**Current LiteLLM Instance:**

- **Local URL**: `http://localhost:4000` (if running locally)

## 🔐 Authentication

### Master Key (Admin Operations)
- Used for: Creating users, managing keys, viewing usage
- Header: `x-litellm-api-key: <master-key>`
- Format: Must start with `sk-`
- **Current Master Key**: `sk-dcb0c8a73a664a2307b8e2f12ef90a34819204105f32f51cc8c621ebf88c7642`

### User API Key (Regular Operations)
- Used for: Making LLM requests
- Header: `Authorization: Bearer <user-api-key>`
- Format: Generated by LiteLLM when user is created
- Obtained from: `/user/info` endpoint

## 💻 Frontend Integration Examples

### React/Next.js Example

```typescript
// lib/litellm-client.ts
const LITELLM_API_URL = process.env.NEXT_PUBLIC_LITELLM_API_URL || 
  'https://swzissb82u.us-east-1.awsapprunner.com';

interface ChatMessage {
  role: 'user' | 'assistant' | 'system';
  content: string;
}

export async function chatCompletion(
  apiKey: string,
  messages: ChatMessage[],
  model: string = 'gemini-2.5-pro'
) {
  const response = await fetch(`${LITELLM_API_URL}/v1/chat/completions`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${apiKey}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      model,
      messages,
      temperature: 0.7,
      max_tokens: 1000,
    }),
  });

  if (!response.ok) {
    const error = await response.text();
    throw new Error(`LiteLLM API error: ${error}`);
  }

  const data = await response.json();
  return data.choices[0].message.content;
}

// Usage in component
export function ChatComponent() {
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [apiKey, setApiKey] = useState<string>('');

  const sendMessage = async (userMessage: string) => {
    const newMessages = [...messages, { role: 'user' as const, content: userMessage }];
    setMessages(newMessages);

    try {
      const response = await chatCompletion(apiKey, newMessages);
      setMessages([...newMessages, { role: 'assistant' as const, content: response }]);
    } catch (error) {
      console.error('Chat error:', error);
    }
  };

  return (
    <div>
      <input 
        type="password" 
        placeholder="Enter LiteLLM API Key"
        value={apiKey}
        onChange={(e) => setApiKey(e.target.value)}
      />
      {/* Chat UI */}
    </div>
  );
}
```

### Vanilla JavaScript Example

```javascript
// litellm-client.js
const LITELLM_API_URL = 'https://swzissb82u.us-east-1.awsapprunner.com';

class LiteLLMClient {
  constructor(apiKey) {
    this.apiKey = apiKey;
    this.baseURL = LITELLM_API_URL;
  }

  async chat(messages, model = 'gemini-2.5-pro') {
    const response = await fetch(`${this.baseURL}/v1/chat/completions`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model,
        messages,
        temperature: 0.7,
      }),
    });

    if (!response.ok) {
      throw new Error(`API Error: ${response.statusText}`);
    }

    return await response.json();
  }

  async getUserInfo(userId, masterKey) {
    const response = await fetch(
      `${this.baseURL}/user/info?user_id=${userId}`,
      {
        headers: {
          'x-litellm-api-key': masterKey,
        },
      }
    );

    if (!response.ok) {
      throw new Error(`Failed to get user info: ${response.statusText}`);
    }

    return await response.json();
  }
}

// Usage
const client = new LiteLLMClient('user-api-key-here');

client.chat([
  { role: 'user', content: 'Hello, how are you?' }
]).then(response => {
  console.log(response.choices[0].message.content);
});
```

### Python Backend Integration

```python
# litellm_client.py
import requests
import os

LITELLM_API_URL = os.getenv('LITELLM_API_URL', 'https://swzissb82u.us-east-1.awsapprunner.com')

class LiteLLMClient:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = LITELLM_API_URL
    
    def chat_completion(self, messages: list, model: str = 'gemini-2.5-pro'):
        """Make a chat completion request"""
        response = requests.post(
            f'{self.base_url}/v1/chat/completions',
            headers={
                'Authorization': f'Bearer {self.api_key}',
                'Content-Type': 'application/json',
            },
            json={
                'model': model,
                'messages': messages,
                'temperature': 0.7,
            }
        )
        response.raise_for_status()
        return response.json()
    
    def get_user_info(self, user_id: str, master_key: str):
        """Get user information (requires master key)"""
        response = requests.get(
            f'{self.base_url}/user/info',
            params={'user_id': user_id},
            headers={'x-litellm-api-key': master_key}
        )
        response.raise_for_status()
        return response.json()

# Usage
client = LiteLLMClient('user-api-key-here')
response = client.chat_completion([
    {'role': 'user', 'content': 'Hello!'}
])
print(response['choices'][0]['message']['content'])
```

## 👥 User Management

### Creating Users from Your Backend

```typescript
// Backend API route (Next.js example)
// app/api/create-litellm-user/route.ts
import { NextRequest, NextResponse } from 'next/server';

const LITELLM_API_URL = process.env.LITELLM_API_URL || 
  'https://swzissb82u.us-east-1.awsapprunner.com';
const LITELLM_MASTER_KEY = process.env.LITELLM_API_KEY || '';

export async function POST(request: NextRequest) {
  const { userId, email, name } = await request.json();

  const response = await fetch(`${LITELLM_API_URL}/user/new`, {
    method: 'POST',
    headers: {
      'x-litellm-api-key': LITELLM_MASTER_KEY,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      user_id: userId.toString(),
      user_email: email,
      user_alias: name,
      auto_create_key: true,
      tpm_limit: 100000,
      rpm_limit: 100,
    }),
  });

  if (!response.ok) {
    return NextResponse.json(
      { error: 'Failed to create LiteLLM user' },
      { status: response.status }
    );
  }

  const data = await response.json();
  return NextResponse.json({ 
    user_id: data.user_id,
    api_key: data.key, // User's API key
  });
}
```

### Streaming Responses

```typescript
async function streamChat(apiKey: string, messages: any[]) {
  const response = await fetch(
    'https://swzissb82u.us-east-1.awsapprunner.com/v1/chat/completions',
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'gemini-2.5-pro',
        messages,
        stream: true,
      }),
    }
  );

  const reader = response.body?.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader!.read();
    if (done) break;

    const chunk = decoder.decode(value);
    const lines = chunk.split('\n').filter(line => line.trim() !== '');

    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const data = line.slice(6);
        if (data === '[DONE]') return;
        
        try {
          const json = JSON.parse(data);
          const content = json.choices[0]?.delta?.content;
          if (content) {
            // Handle streaming content
            console.log(content);
          }
        } catch (e) {
          // Skip invalid JSON
        }
      }
    }
  }
}
```

## ⚠️ Error Handling

```typescript
async function handleLLMRequest(apiKey: string, messages: any[]) {
  try {
    const response = await fetch(
      'https://swzissb82u.us-east-1.awsapprunner.com/v1/chat/completions',
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          model: 'gemini-2.5-pro',
          messages,
        }),
      }
    );

    if (!response.ok) {
      const error = await response.json().catch(() => ({ 
        error: await response.text() 
      }));
      
      // Handle specific error codes
      if (response.status === 401) {
        throw new Error('Invalid API key');
      } else if (response.status === 429) {
        throw new Error('Rate limit exceeded');
      } else if (response.status === 500) {
        throw new Error('LiteLLM server error');
      } else {
        throw new Error(error.error || 'Unknown error');
      }
    }

    return await response.json();
  } catch (error) {
    if (error instanceof TypeError) {
      // Network error
      throw new Error('Network error: Could not connect to LiteLLM');
    }
    throw error;
  }
}
```

## ✅ Best Practices

### 1. **Never Expose Master Key in Frontend**
   - Master key should ONLY be used in backend/server-side code
   - Frontend should only use user API keys

### 2. **Store User API Keys Securely**
   ```typescript
   // ✅ Good: Store in secure storage
   localStorage.setItem('litellm_api_key', userApiKey);
   
   // ❌ Bad: Don't hardcode keys
   const apiKey = 'sk-...'; // Never do this!
   ```

### 3. **Handle Rate Limits**
   ```typescript
   async function chatWithRetry(apiKey: string, messages: any[], retries = 3) {
     for (let i = 0; i < retries; i++) {
       try {
         return await chatCompletion(apiKey, messages);
       } catch (error) {
         if (error.message.includes('rate limit') && i < retries - 1) {
           await new Promise(resolve => setTimeout(resolve, 1000 * (i + 1)));
           continue;
         }
         throw error;
       }
     }
   }
   ```

### 4. **Monitor Usage**
   ```typescript
   // Periodically check user usage
   async function checkUsage(userId: number) {
     const userInfo = await getUserInfo(userId, masterKey);
     console.log('Spend:', userInfo.spend);
     console.log('Tokens used:', userInfo.total_tokens);
     console.log('TPM Limit:', userInfo.tpm_limit);
   }
   ```

### 5. **Environment Variables**
